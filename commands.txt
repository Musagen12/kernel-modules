Port fowarding
---------------
qemu-system-x86_64 \
  -kernel bzImage \
  -drive file=rootfs.img,format=raw \
  -append "console=ttyS0 nokaslr net.ifnames=0 root=/dev/sda rw" \
  -nographic \
  -m 2G \
  -netdev user,id=net0,hostfwd=tcp::2222-:22 \
  -device e1000,netdev=net0 \
  -serial mon:stdio


Copying files into a qemu vm
-----------------------------
scp -P 2222 driver.ko root@localhost:/root/

SSH connection into the vm
---------------------------
ssh -p 2222 root@localhost
